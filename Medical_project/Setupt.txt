#setting up envi
# 1. Create a safe sandbox for your work
python -m venv /workspace/unsloth_env
source /workspace/unsloth_env/bin/activate

# 2. Install Blackwell-specific Torch (Nightly supports SM 10.0)
pip install --pre torch torchvision torchaudio

# 3. Install Unsloth and essential tools
pip install "unsloth[colab-new] @ git+https://github.com/unslothai/unsloth.git"
pip install --no-deps xformers trl peft accelerate bitsandbytes sentencepiece
pip install transformers
pip install qdrant-client	
pip install einops
pip install timm

# 4. Make this environment show up in Jupyter
pip install gguf protobuf sentencepiece mistral_common ipykernel
python -m ipykernel install --user --name=unsloth_env --display-name "Unsloth (RTX 5090)"
